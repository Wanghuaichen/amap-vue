# 安装使用

# 引入依赖

yarn add @amap/amap-vue # 或 npm install --save @amap/amap-vue
1

# 在 Vue 中使用

# 全局加载

在你的 Vue 项目入口文件（例如main.js）里全局引入

import AmapVue from '@amap/amap-vue'

AmapVue.config.version = '1.4.15' // 默认1.4.15
AmapVue.config.key = '您的JSAPI KEY'
AmapVue.config.plugins = [
  'AMap.ToolBar',
  // 在此配置需要预加载的插件，如果不配置，则会在调用的时候再自动加载
]
Vue.use(AmapVue)

# 按需引入

在你的 Vue 项目入口文件（例如main.js）里初始化配置

// main.js
import AmapVueConfig from '@amap/amap-vue/lib/config'
AmapVueConfig.key = '您的JSAPI KEY'
1
2
3

在你的 Vue 组件里按需引入组件

<script>
// your-component.vue
import Amap from '@amap/amap-vue/lib/components/Amap'
import AmapMarker from '@amap/amap-vue/lib/components/overlay/Marker'

export default {
  components: {
    Amap,
    AmapMarker,
  },
}
</script>
